```swift
open class Interceptor: RequestInterceptor {
    public let adapters: [any RequestAdapter]
    public let retriers: [any RequestRetrier]
    
    open func adapt(_ urlRequest: URLRequest, for session: Session, completion: @escaping (Result<URLRequest, any Error>) -> Void) {
        adapt(urlRequest, for: session, using: adapters, completion: completion)
    }
    
    private func adapt(_ urlRequest: URLRequest,
                       for session: Session,
                       using adapters: [any RequestAdapter],
                       completion: @escaping (Result<URLRequest, any Error>) -> Void) {
        var pendingAdapters = adapters
        guard !pendingAdapters.isEmpty else { completion(.success(urlRequest)); return }
        
        let adapter = pendingAdapters.removeFirst()
        adapter.adapt(urlRequest, for: session) { result in
            switch result {
            case let .success(urlRequest):
                self.adapt(urlRequest, for: session, using: pendingAdapters, completion: completion)
            case .failure:
                completion(result)
            }
        }
    }
}
```

Adapters execute sequentially; first failure stops the chain.

### 2b. URL Request Creation and Adaptation

**File:** `Source/Core/Session.swift` (lines ~1246-1310)

```swift
func performSetupOperations(for request: Request,
                            convertible: any URLRequestConvertible,
                            shouldCreateTask: @escaping @Sendable () -> Bool = { true }) {
    dispatchPrecondition(condition: .onQueue(requestQueue))

    let initialRequest: URLRequest
    do {
        initialRequest = try convertible.asURLRequest()  // Convert (e.g., RequestConvertible.asURLRequest)
        try initialRequest.validate()                     // Basic validation
    } catch {
        rootQueue.async { request.didFailToCreateURLRequest(with: error.asAFError(...)) }
        return
    }

    rootQueue.async { request.didCreateInitialURLRequest(initialRequest) }
    guard !request.isCancelled else { return }

    // Get adapter (Session + Request interceptors merged)
    guard let adapter = adapter(for: request) else {
        guard shouldCreateTask() else { return }
        rootQueue.async { self.didCreateURLRequest(initialRequest, for: request) }
        return
    }

    let adapterState = RequestAdapterState(requestID: request.id, session: self)
    adapter.adapt(initialRequest, using: adapterState) { result in
        do {
            let adaptedRequest = try result.get()
            try adaptedRequest.validate()
            
            self.rootQueue.async { request.didAdaptInitialRequest(initialRequest, to: adaptedRequest) }
            guard shouldCreateTask() else { return }
            
            self.rootQueue.async { self.didCreateURLRequest(adaptedRequest, for: request) }
        } catch {
            self.rootQueue.async { request.didFailToAdaptURLRequest(initialRequest, withError: .requestAdaptationFailed(error: error)) }
        }
    }
}

func adapter(for request: Request) -> (any RequestAdapter)? {
    if let requestInterceptor = request.interceptor, let sessionInterceptor = interceptor {
        Interceptor(adapters: [sessionInterceptor, requestInterceptor])  // Session adapters first
    } else {
        request.interceptor ?? interceptor
    }
}
```

1. **Initial URLRequest**: `convertible.asURLRequest()` (from `RequestConvertible` which calls `encoding.encode(...)`)

```swift
    dispatchPrecondition(condition: .onQueue(underlyingQueue))

    mutableState.write { mutableState in
        mutableState.tasks.append(task)

        if let urlSessionTaskHandler = mutableState.urlSessionTaskHandler {
            urlSessionTaskHandler.queue.async { urlSessionTaskHandler.handler(task) }

        mutableState.eventMonitor?.request(self, didCreateTask: task)

        // Resume/suspend/cancel task based on current state
        switch mutableState.state {
        case .initialized, .finished:

        case .cancelled:

            task.cancel()
            underlyingQueue.async { self.didCancelTask(task) }
        }
    }
```

### 3f. Response Handler Chain

**File:** `Source/Core/Request.swift` (lines ~646-700)

```swift
func appendResponseSerializer(_ closure: @escaping @Sendable () -> Void) {
    mutableState.write { mutableState in
        mutableState.responseSerializers.append(closure)

        if mutableState.state == .finished {
            mutableState.state = .resumed
        }

        // If serializers have already been processed, execute immediately
        if mutableState.responseSerializerProcessingFinished {
            underlyingQueue.async { self.processNextResponseSerializer() }
        }

        if mutableState.state.canTransitionTo(.resumed) {
            underlyingQueue.async { [self] in
                if (shouldAutomaticallyResume ?? delegate?.startImmediately) == true {
                    resume()
                }
            }
        }
    }
}

func processNextResponseSerializer() {
    let executeOutside: (() -> Void)? = mutableState.write { mutableState in
        guard !mutableState.isResponseSerializerEnqueued else { return nil }

        let responseSerializerIndex = mutableState.responseSerializerCompletions.count
        let isAvailableSerializer = responseSerializerIndex < mutableState.responseSerializers.count
        let responseSerializer = isAvailableSerializer ? mutableState.responseSerializers[responseSerializerIndex] : nil

        if let responseSerializer {
            mutableState.isResponseSerializerEnqueued = true
            return { self.serializationQueue.async { responseSerializer() } }
        } else {
            let completions = mutableState.responseSerializerCompletions
            mutableState.responseSerializers.removeAll()
            mutableState.responseSerializerCompletions.removeAll()

            if mutableState.state.canTransitionTo(.finished) {
                mutableState.state = .finished
            }

            mutableState.responseSerializerProcessingFinished = true
            mutableState.isFinishing = false

            return {
                completions.forEach { $0() }
                self.cleanup()
            }
        }
    }

    executeOutside?()
}
```

Response serializers execute sequentially on `serializationQueue`, then completions are called on the user's specified queue (e.g., `.main`).

```
1. BUILD
  Session.request(...) 
    → RequestConvertible/RequestEncodableConvertible.asURLRequest()
      → ParameterEncoding/ParameterEncoder.encode(...)
    → DataRequest created (state: .initialized)

2. SEND
  perform(request)
    → requestQueue.async { performDataRequest(...) }
      → performSetupOperations(...)
        → convertible.asURLRequest() [initial]
        → adapter.adapt(...) [if present]
      → didCreateURLRequest(adapted, for: request)
        → request.task(for:using:) → URLSessionTask
        → mutableState.requestTaskMap[request] = task
        → task.resume() [if state == .resumed]
  
  URLSessionDelegate callbacks (on rootQueue via SessionDelegate)
    → didReceive(data:)
    → didReceiveResponse(...)
    → didFinishCollecting(metrics:)
    → didCompleteWithError(...)
      → request.didCompleteTask(task, with: error)
        → validators.forEach { validator() } [synchronous validation]
        → retryOrFinish(error:)

3. VALIDATE
  validate(statusCode:) / validate(contentType:) / validate(:)
    → Closure appended to request.validators array
  
  When didCompleteTask called:
    → All validators execute synchronously on underlyingQueue
    → If ValidationResult.failure → error set on request

    → request.finish()
      → processNextResponseSerializer()
      → Response handlers run on serializationQueue then user's queue
```

## 1. Build — `Session.request(...)` → a `DataRequest`

**Entry point** (`Source/Core/Session.swift`): the convenience overloads take a `URLConvertible` plus `method`/`parameters`/`encoding` (or an `encoder` for `Encodable` parameters), an optional `RequestInterceptor`, and a `requestModifier`.

Rather than building the `URLRequest` eagerly, `Session` wraps your inputs in one of two internal `URLRequestConvertible` structs and defers the work:

```swift
struct RequestConvertible: URLRequestConvertible {
    func asURLRequest() throws -> URLRequest {
        var request = try URLRequest(url: url, method: method, headers: headers)
        try requestModifier?(&request)
        return try encoding.encode(request, with: parameters)   // URLEncoding/JSONEncoding
    }
}
struct RequestEncodableConvertible<P: Encodable & Sendable>: URLRequestConvertible {
    func asURLRequest() throws -> URLRequest {
        var request = try URLRequest(url: url, method: method, headers: headers)
        try requestModifier?(&request)
        return try parameters.map { try encoder.encode($0, into: request) } ?? request
    }
}
```

- **`URLConvertible`/`URLRequestConvertible`** (`URLConvertible+URLRequestConvertible.swift`): `String`/`URL`/`URLComponents` produce a `URL`; `URLRequest` produces itself.
- **Parameter encoding**: `ParameterEncoding` (`URLEncoding`, `JSONEncoding`) for `[String: Any]` dictionaries — `URLEncoding` decides query-string vs. httpBody by HTTP method; `ParameterEncoder` (`URLEncodedFormParameterEncoder`, `JSONParameterEncoder`) for `Encodable` types, setting `Content-Type` and `httpBody`.

`Session` then constructs the `DataRequest` (starting in state `.initialized`), wiring in `rootQueue`, `serializationQueue`, the event monitor, the interceptor, and itself as `delegate`. Note the actual `URLRequest` doesn't exist yet — it's still a lazy `convertible`.

## 2. Send — `perform` → adapt → `URLSessionTask`

`Session.perform(_:)` hops onto `rootQueue` (the single serial queue guarding all session state), registers the request in `activeRequests`, then dispatches onto `requestQueue` to `performDataRequest`. The queue design — `rootQueue` (state), `requestQueue` (URLRequest creation), `serializationQueue` (response parsing) — all target `rootQueue` by default for serialized safety.

`performSetupOperations` is where the lazy `convertible` finally materializes:

1. `initialRequest = try convertible.asURLRequest()` then `initialRequest.validate()` — this runs the encoding from step 1.
2. The **adapt** step: `adapter(for:)` merges session + request interceptors (`Interceptor(adapters: [sessionInterceptor, requestInterceptor])`, session first) and calls `adapter.adapt(...)`, which runs adapters sequentially — first failure aborts the chain (e.g. injecting auth headers). Both initial and adapted `URLRequest`s are retained in `request.requests`.
3. `didCreateURLRequest` calls `request.task(for:using:)` → `session.dataTask(with:)`, stores it in **`RequestTaskMap`** (a bidirectional `Request ↔ URLSessionTask` map so delegate callbacks can find their request), and resumes the task if the request's state is `.resumed`.

The **`Request` state machine** (`Request.swift`) — `initialized → resumed/suspended → cancelled/finished` — gates `task.resume()/suspend()/cancel()`; all mutable state lives in a lock-guarded `Protected<MutableState>`.

**`SessionDelegate`** (`URLSessionDataDelegate`) drives the live request: `didReceive(data:)` appends into `DataRequest`'s accumulating buffer, `didFinishCollecting metrics` records metrics, and `didCompleteWithError` routes through the task map back to `request.didCompleteTask(...)`.

**Retry**: on completion with an error, `retryOrFinish` asks the `RequestRetrier` via `delegate.retryResult(for:dueTo:)`. The `RetryResult` (`.retry`, `.retryWithDelay`, `.doNotRetry`, `.doNotRetryWithError`) decides whether to re-`perform` or `finish`.

## 3. Validate — closures run at completion, before serialization

`DataRequest.validate(...)` (`Source/Features/Validation.swift`) appends a closure to a `Protected` `validators` array and returns `Self` for chaining. Bare `validate()` = acceptable status codes (2xx) + content types from the `Accept` header:

```swift
public typealias Validation = @Sendable (URLRequest?, HTTPURLResponse, Data?) -> ValidationResult
public typealias ValidationResult = Result<Void, any (Error & Sendable)>
```

Built-in checks: `unacceptableStatusCode` when the code isn't in the acceptable set, and content-type checks via a `MIMEType` matcher that honors wildcards (`*/*`, `type/*`). Failures become `AFError.responseValidationFailed(reason:)` (`ResponseValidationFailureReason`: `unacceptableStatusCode`, `unacceptableContentType`, `missingContentType`, `customValidationFailed`, …).

Crucially, validators run **inside `didCompleteTask`**, synchronously on `underlyingQueue`, *before* retry/finish:

```swift
func didCompleteTask(_ task: URLSessionTask, with error: AFError?) {
    mutableState.write { $0.error = $0.error ?? error }
    validators.read(\.self).forEach { $0() }   // each sets request.error on failure
    retryOrFinish(error: self.error)            // validation error can trigger a retry
}
```

Because validation populates `request.error` before `retryOrFinish`, a failed validation can itself drive the retrier. If the request finishes, `processNextResponseSerializer` runs each response serializer (`response`, `responseDecodable`, …) sequentially on `serializationQueue`, then fires the completion handler on the user's queue (default `.main`) with a `DataResponse` carrying the `result` (`.success`/`.failure(AFError)`).

**Flow in one line:** `request(...)` wraps inputs in a lazy `URLRequestConvertible` → `perform` runs encoding + adapter chain → builds & resumes a `URLSessionTask` tracked in `RequestTaskMap` → `SessionDelegate` accumulates data and signals completion → validators run, then retry-or-finish, then serializers deliver a typed `DataResponse`.
