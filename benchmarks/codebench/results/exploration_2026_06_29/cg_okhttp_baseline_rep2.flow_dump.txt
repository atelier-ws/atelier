```kotlin
  interceptors += client.interceptors                    // Application interceptors
  interceptors += RetryAndFollowUpInterceptor()         // 1. Retry & Redirects
  interceptors += BridgeInterceptor()                    // 2. Bridge

  }
  interceptors += CallServerInterceptor                 // 5. Call Server (final)

  val chain =
    RealInterceptorChain(
      call = this,
      interceptors = interceptors,
      index = 0,
      exchange = null,
      request = originalRequest,
    )

  var calledNoMoreExchanges = false
  try {
    val response = chain.proceed(originalRequest)
    if (isCanceled()) {
      response.closeQuietly()
      throw IOException("Canceled")
    }
    return response
  } catch (e: IOException) {
```

### 3. **RealInterceptorChain (the proceed method and chain advancement)**
**File:** `/tmp/codebench_workspaces/20260629T234322Z-36eed4b76d68/cg_okhttp_baseline_rep2/okhttp/src/commonJvmAndroid/kotlin/okhttp3/internal/http/RealInterceptorChain.kt`

```kotlin
class RealInterceptorChain(
  internal val call: RealCall,
  private val interceptors: List<Interceptor>,
  private val index: Int,
  internal val exchange: Exchange?,
  internal val request: Request,
  // ... timeout and other configuration fields
) : Interceptor.Chain {
  private var calls: Int = 0

  @Throws(IOException::class)
  override fun proceed(request: Request): Response {
    check(index < interceptors.size)

    calls++

    if (exchange != null) {
      check(exchange.finder.routePlanner.sameHostAndPort(request.url)) {
        "network interceptor ${interceptors[index - 1]} must retain the same host and port"
      }
      check(calls == 1) {
        "network interceptor ${interceptors[index - 1]} must call proceed() exactly once"
      }
    }

    // Call the next interceptor in the chain.
    val next = copy(index = index + 1, request = request)
    val interceptor = interceptors[index]

    @Suppress("USELESS_ELVIS")
    val response =
      interceptor.intercept(next) ?: throw NullPointerException(
        "interceptor $interceptor returned null",
      )

    if (exchange != null) {
      check(index + 1 >= interceptors.size || next.calls == 1) {
        "network interceptor $interceptor must call proceed() exactly once"
      }
    }

    return response
  }
}
```

**Key points:**
- `proceed()` validates the index is within bounds (`index < interceptors.size`)
- Each call to `proceed()` increments a counter to detect recursive calls
- It creates a new chain with `index + 1` by calling `copy(index = index + 1, request = request)` 
- Gets the interceptor at the current index: `interceptors[index]`
- Calls `interceptor.intercept(next)` to invoke the interceptor with the advanced chain
- When the interceptor calls `chain.proceed()`, it's calling the *new* chain (with index + 1), which will invoke the next interceptor
- This continues until the last interceptor (CallServerInterceptor) makes the actual network request
- Network interceptors are validated to: maintain the same host/port, and call proceed() exactly once

```kotlin
class RetryAndFollowUpInterceptor : Interceptor {
  @Throws(IOException::class)
  override fun intercept(chain: Interceptor.Chain): Response {
    val realChain = chain as RealInterceptorChain
    var request = chain.request
    val call = realChain.call
    var followUpCount = 0
    var priorResponse: Response? = null
    var newRoutePlanner = true
    var recoveredFailures = listOf<IOException>()
    while (true) {
      call.enterNetworkInterceptorExchange(request, newRoutePlanner, chain)

      var response: Response
      var closeActiveExchange = true
      try {
        if (call.isCanceled()) {
          throw IOException("Canceled")
        }

        try {
          response = realChain.proceed(request)
          newRoutePlanner = true
        } catch (e: IOException) {
          val isRecoverable = recover(e, call, chain, request)
          call.eventListener.retryDecision(call, e, isRecoverable)
          if (!isRecoverable) throw e.withSuppressed(recoveredFailures)
          recoveredFailures += e
          newRoutePlanner = false
          continue
        }

        // Process response for follow-ups (redirects, auth challenges, etc.)
        response = response.newBuilder().request(request).priorResponse(priorResponse?.stripBody()).build()

        val exchange = call.interceptorScopedExchange
        val followUp = followUpRequest(response, exchange, chain)

        if (followUp == null) {
          if (exchange != null && exchange.isDuplex) {
            call.timeoutEarlyExit()
          }
          closeActiveExchange = false
          call.eventListener.followUpDecision(call, response, null)
          return response  // No more follow-ups, return response
        }

        // Continue loop with follow-up request
        request = followUp
        priorResponse = response
      } finally {
        call.exitNetworkInterceptorExchange(closeActiveExchange)
      }
    }
  }

  private fun recover(e: IOException, call: RealCall, chain: Interceptor.Chain, userRequest: Request): Boolean {
    // Check: retryOnConnectionFailure enabled?
    if (!chain.retryOnConnectionFailure) return false
    // Check: can retry this request body?
    if (requestSendStarted && requestIsOneShot(e, userRequest)) return false
    // Check: is this exception recoverable?
    if (!isRecoverable(e, requestSendStarted)) return false
    // Check: are there more routes to try?
    if (!call.retryAfterFailure()) return false
    return true
  }
}
```

**Purpose:** Recovers from failures and follows redirects. Uses a loop to retry failed requests and handle HTTP redirects (3xx), authentication (401/407), and client timeouts (408/503).

```kotlin
class BridgeInterceptor : Interceptor {
  @Throws(IOException::class)
  override fun intercept(chain: Interceptor.Chain): Response {
    val userRequest = chain.request()
    val requestBuilder = userRequest.newBuilder()

    // Add Content-Type, Content-Length, or Transfer-Encoding
    val body = userRequest.body
    if (body != null) {
      val contentType = body.contentType()
      if (contentType != null) {
        requestBuilder.header("Content-Type", contentType.toString())
      }
      val contentLength = body.contentLength()
      if (contentLength != -1L) {
        requestBuilder.header("Content-Length", contentLength.toString())
        requestBuilder.removeHeader("Transfer-Encoding")
      } else {
        requestBuilder.header("Transfer-Encoding", "chunked")
        requestBuilder.removeHeader("Content-Length")
      }
    }

    // Add Host header
    if (userRequest.header("Host") == null) {
      requestBuilder.header("Host", userRequest.url.toHostHeader())
    }

    // Add Connection header
    if (userRequest.header("Connection") == null) {
      requestBuilder.header("Connection", "Keep-Alive")
    }

    // Add Accept-Encoding: gzip if not present and no Range header
    var transparentGzip = false
    if (userRequest.header("Accept-Encoding") == null && userRequest.header("Range") == null) {
      transparentGzip = true
      requestBuilder.header("Accept-Encoding", "gzip")
    }

    // Add cookies
    val cookies = chain.cookieJar.loadForRequest(userRequest.url)
    if (cookies.isNotEmpty()) {
      requestBuilder.header("Cookie", cookieHeader(cookies))
    }

    // Add User-Agent
    if (userRequest.header("User-Agent") == null) {
      requestBuilder.header("User-Agent", USER_AGENT)
    }

    val networkRequest = requestBuilder.build()
    val networkResponse = chain.proceed(networkRequest)

    // Save cookies from response
    chain.cookieJar.receiveHeaders(networkRequest.url, networkResponse.headers)

    val responseBuilder = networkResponse.newBuilder().request(networkRequest)

    // Handle transparent gzip decompression
    if (transparentGzip && "gzip".equals(networkResponse.header("Content-Encoding"), ignoreCase = true) && 
        networkResponse.promisesBody()) {
      val responseBody = networkResponse.body
      val gzipSource = GzipSource(responseBody.source())
      val strippedHeaders = networkResponse.headers.newBuilder()
        .removeAll("Content-Encoding")
        .removeAll("Content-Length")
        .build()
      responseBuilder.headers(strippedHeaders)
      val contentType = networkResponse.header("Content-Type")
      responseBuilder.body(RealResponseBody(contentType, -1L, gzipSource.buffer()))
    }

    return responseBuilder.build()
  }
}
```

**Purpose:** Bridges from application code to network code. Adds standard HTTP headers (Content-Type, Host, Connection, User-Agent, cookies, Accept-Encoding) and handles transparent gzip decompression of responses.

#### **3. CacheInterceptor**
**File:** `/tmp/codebench_workspaces/20260629T234322Z-36eed4b76d68/cg_okhttp_baseline_rep2/okhttp/src/commonJvmAndroid/kotlin/okhttp3/internal/cache/CacheInterceptor.kt`

```kotlin
class CacheInterceptor : Interceptor {
  @Throws(IOException::class)
  override fun intercept(chain: Interceptor.Chain): Response {
    val call = chain.call()
    val cache = chain.cache
    val cacheCandidate = cache?.get(chain.request().requestForCache())

    val now = System.currentTimeMillis()
    val strategy = CacheStrategy.Factory(now, chain.request(), cacheCandidate).compute()
    val networkRequest = strategy.networkRequest
    val cacheResponse = strategy.cacheResponse

    cache?.trackResponse(strategy)

    // Close cache candidate if not applicable
    if (cacheCandidate != null && cacheResponse == null) {
      cacheCandidate.body.closeQuietly()
    }

    // If forbidden from network and no cache, return 504
    if (networkRequest == null && cacheResponse == null) {
      return Response.Builder()
        .request(chain.request())
        .protocol(Protocol.HTTP_1_1)
        .code(HTTP_GATEWAY_TIMEOUT)
        .message("Unsatisfiable Request (only-if-cached)")
        .sentRequestAtMillis(-1L)
        .receivedResponseAtMillis(System.currentTimeMillis())
        .build()
    }

    // If no network request needed, return cached response
    if (networkRequest == null) {
      return cacheResponse!!.newBuilder().cacheResponse(cacheResponse.stripBody()).build()
    }

    // If cache exists, we might do a conditional GET (304 Not Modified)
    if (cacheResponse != null) {
      chain.eventListener.cacheConditionalHit(call, cacheResponse)
    } else if (cache != null) {
      chain.eventListener.cacheMiss(call)
    }

    // Make the network request
    var networkResponse: Response? = null
    try {
      networkResponse = chain.proceed(networkRequest)
    } finally {
      if (networkResponse == null && cacheCandidate != null) {
        cacheCandidate.body.closeQuietly()
      }
    }

    // If 304 Not Modified and we have a cached response, combine and return cached version
    if (cacheResponse != null) {
      if (networkResponse?.code == HTTP_NOT_MODIFIED) {
        val response = cacheResponse.newBuilder()
          .headers(combine(cacheResponse.headers, networkResponse.headers))
          .sentRequestAtMillis(networkResponse.sentRequestAtMillis)
          .receivedResponseAtMillis(networkResponse.receivedResponseAtMillis)
          .cacheResponse(cacheResponse.stripBody())
          .networkResponse(networkResponse.stripBody())
          .build()

        networkResponse.body.close()
        cache!!.trackConditionalCacheHit()
        cache.update(cacheResponse, response)
        return response
      } else {
        cacheResponse.body.closeQuietly()
      }
    }

    // Prepare response and possibly cache it
    val response = networkResponse!!.newBuilder()
      .cacheResponse(cacheResponse?.stripBody())
      .networkResponse(networkResponse.stripBody())
      .build()

    if (cache != null) {
      if (response.promisesBody() && CacheStrategy.isCacheable(response, networkRequest)) {
        val cacheRequest = cache.put(response.newBuilder().request(networkRequest).build())
        return cacheWritingResponse(cacheRequest, response)
      }

      if (HttpMethod.invalidatesCache(networkRequest.method)) {
        try {
          cache.remove(networkRequest)
        } catch (_: IOException) {
          // The cache cannot be written.
        }
      }
    }

    return response
  }
}
```

**Purpose:** Serves requests from cache and writes responses to cache. Implements HTTP cache semantics: returns cached responses when fresh, performs conditional GET (304 Not Modified) when needed, and caches appropriate responses.

```kotlin
    val realChain = chain as RealInterceptorChain
```

```kotlin
object CallServerInterceptor : Interceptor {
  @Throws(IOException::class)
  override fun intercept(chain: Interceptor.Chain): Response {
    val realChain = chain as RealInterceptorChain
    val exchange = realChain.exchange!!
    val request = realChain.request
    val requestBody = request.body
    val sentRequestMillis = System.currentTimeMillis()

    var invokeStartEvent = true
    var responseBuilder: Response.Builder? = null
    var sendRequestException: IOException? = null
    val hasRequestBody = HttpMethod.permitsRequestBody(request.method) && requestBody != null
    val isUpgradeRequest = "upgrade".equals(request.header("Connection"), ignoreCase = true)

    try {
      // Write request headers
      exchange.writeRequestHeaders(request)

      if (hasRequestBody) {
        // Handle "Expect: 100-continue"
        if ("100-continue".equals(request.header("Expect"), ignoreCase = true)) {
          exchange.flushRequest()
          responseBuilder = exchange.readResponseHeaders(expectContinue = true)
          exchange.responseHeadersStart()
          invokeStartEvent = false
        }
        
        if (responseBuilder == null) {
          if (requestBody.isDuplex()) {
            exchange.flushRequest()
            val bufferedRequestBody = exchange.createRequestBody(request, true).buffer()
            requestBody.writeTo(bufferedRequestBody)
          } else {
            val bufferedRequestBody = exchange.createRequestBody(request, false).buffer()
            requestBody.writeTo(bufferedRequestBody)
            bufferedRequestBody.close()
          }
        } else {
          exchange.noRequestBody()
          if (!exchange.connection.isMultiplexed) {
            exchange.noNewExchangesOnConnection()
          }
        }
      } else {
        exchange.noRequestBody()
      }

      if (requestBody == null || !requestBody.isDuplex()) {
        exchange.finishRequest()
      }
    } catch (e: IOException) {
      if (e is ConnectionShutdownException) {
        throw e
      }
      if (!exchange.hasFailure) {
        throw e
      }
      sendRequestException = e
    }

    try {
      if (responseBuilder == null) {
        responseBuilder = exchange.readResponseHeaders(expectContinue = false)!!
        if (invokeStartEvent) {
          exchange.responseHeadersStart()
          invokeStartEvent = false
        }
      }

      var response = responseBuilder
        .request(request)
        .handshake(exchange.connection.handshake())
        .sentRequestAtMillis(sentRequestMillis)
        .receivedResponseAtMillis(System.currentTimeMillis())
        .build()

      // Handle 1xx informational responses
      while (shouldIgnoreAndWaitForRealResponse(response.code)) {
        responseBuilder = exchange.readResponseHeaders(expectContinue = false)!!
        if (invokeStartEvent) {
          exchange.responseHeadersStart()
        }
        response = responseBuilder
          .request(request)
          .handshake(exchange.connection.handshake())
          .sentRequestAtMillis(sentRequestMillis)
          .receivedResponseAtMillis(System.currentTimeMillis())
          .build()
      }

      exchange.responseHeadersEnd(response)

      // Handle upgrade requests
      if (isUpgradeRequest && code == HTTP_SWITCHING_PROTOCOLS) {
        response = response.newBuilder()
          .body(UnreadableResponseBody(...))
          .socket(exchange.upgradeToSocket())
          .build()
      } else {
        // Normal response: open response body
        val responseBody = exchange.openResponseBody(response)
        response = response.newBuilder()
          .body(responseBody)
          .trailers(...)
          .build()
      }

      if ("close".equals(response.request.header("Connection"), ignoreCase = true) ||
          "close".equals(response.header("Connection"), ignoreCase = true)) {
        exchange.noNewExchangesOnConnection()
      }

      return response
    } catch (e: IOException) {
      if (sendRequestException != null) {
        sendRequestException.addSuppressed(e)
        throw sendRequestException
      }
      throw e
    }
  }

  private fun shouldIgnoreAndWaitForRealResponse(code: Int): Boolean =
    code == 100 || code in (102 until 200)
}
```

**Purpose:** The final interceptor that actually makes the network request. Writes request headers and body to the connection using the Exchange, reads the response headers and body, and handles protocol-level details like "Expect: 100-continue", HTTP upgrades (websockets), and informational 1xx responses.

## Summary of Chain Execution Flow

1. **RealCall.execute()** or **RealCall.enqueue()** initiates the request
2. **getResponseWithInterceptorChain()** builds the interceptor list and creates a **RealInterceptorChain** with index=0
3. **chain.proceed()** is called on the initial chain (index 0):
   - Creates a new chain with index=1 and passes it to the current interceptor
   - Current interceptor (at index 0) calls `chain.proceed()` on the new chain
4. This continues, with each interceptor creating a chain with the next index, until:
   - **CallServerInterceptor** (the last one) makes the actual network call via the Exchange
   - Response flows back through each interceptor in reverse order
5. Each interceptor can:
   - Modify the request before calling `proceed()`
   - Modify or short-circuit the response after `proceed()` returns
   - Wrap or replace the response body

The key insight is that **proceed() advances the chain by incrementing the index**, creating a new chain object that points to the next interceptor, ensuring a clean pass-through from one to the next.

## 1. The trigger — `RealCall`

Whether you call `execute()` (synchronous) or `enqueue()` (asynchronous, runs on a dispatcher thread via `AsyncCall`), both funnel into the same method:

`okhttp3/internal/connection/RealCall.kt`

```kotlin
internal fun getResponseWithInterceptorChain(): Response {
  val interceptors = mutableListOf<Interceptor>()
  interceptors += client.interceptors            // your application interceptors
  interceptors += RetryAndFollowUpInterceptor()  // built-in
  interceptors += BridgeInterceptor()
  interceptors += CacheInterceptor()
  interceptors += ConnectInterceptor
  if (!forWebSocket) interceptors += client.networkInterceptors  // your network interceptors
  interceptors += CallServerInterceptor          // terminal

  val chain = RealInterceptorChain(call = this, interceptors = interceptors,
                                   index = 0, exchange = null, request = originalRequest)
  val response = chain.proceed(originalRequest)
  ...
}
```

The **order matters**: application interceptors wrap everything; network interceptors sit just above the wire. The terminal `CallServerInterceptor` is the only one that doesn't delegate further.

## 2. The engine — `RealInterceptorChain.proceed()`

This is the heart. The chain is an **immutable, index-based linked structure**. Each `proceed()` doesn't loop — it builds a *new* chain pointing at the next index and hands it to the current interceptor:

`okhttp3/internal/http/RealInterceptorChain.kt`

```kotlin
override fun proceed(request: Request): Response {
  check(index < interceptors.size)
  calls++
  ...
  val next = copy(index = index + 1, request = request)  // chain for the NEXT interceptor
  val interceptor = interceptors[index]                   // the CURRENT interceptor
  val response = interceptor.intercept(next) ?: throw NullPointerException(...)
  return response
}
```

So control passes like a recursive descent:

```
proceed(0) → interceptors[0].intercept(chain@1)
                 └─ chain@1.proceed() → interceptors[1].intercept(chain@2)
                        └─ chain@2.proceed() → ... → CallServerInterceptor (no proceed())
```

Each interceptor's `intercept` looks like: *modify request → call `chain.proceed()` → modify response → return*. The `calls` counter and host/port checks enforce that **network interceptors call `proceed()` exactly once** and don't change the destination.

## 3. The `Interceptor` contract

`okhttp3/Interceptor.kt` — a functional interface:

```kotlin
fun interface Interceptor {
  fun intercept(chain: Chain): Response
  interface Chain {
    fun request(): Request
    fun proceed(request: Request): Response
    ...
  }
}
```

An interceptor that never calls `proceed()` **short-circuits** the chain (e.g. cache returning a stored response, or a mock).

## 4. What each built-in interceptor does

| # | Interceptor | Role |
|---|-------------|------|
| 1 | **RetryAndFollowUpInterceptor** | Wraps everything in a `while(true)` loop: retries recoverable connection failures (over alternate routes) and follows up on redirects (3xx), auth challenges (401/407), and 408/503. Returns once `followUpRequest` yields `null`. |
| 2 | **BridgeInterceptor** | Translates a user request into a wire request: adds `Content-Type`/`Content-Length`/`Transfer-Encoding`, `Host`, `Connection: Keep-Alive`, `User-Agent`, cookies, and `Accept-Encoding: gzip`. On the way back, saves cookies and **transparently gunzips** the response body. |
| 3 | **CacheInterceptor** | Consults the HTTP cache via `CacheStrategy`. May return a cached response without proceeding, do a conditional GET (and merge a `304 Not Modified` with the cached copy), or write the network response into the cache on the way back. |
| 4 | **ConnectInterceptor** | Acquires/opens a connection: `call.initExchange()` produces an `Exchange` (codec + connection), attaches it to a new chain (`copy(exchange = exchange)`), then proceeds. This is where network interceptors gain access to the live connection. |
| 5 | **CallServerInterceptor** | Terminal — writes request headers/body to the socket and reads the response headers/body via the `Exchange`. Handles `Expect: 100-continue`, 1xx informational responses, and protocol upgrades (WebSocket). Does **not** call `proceed()`. |

## The mental model

The request flows **down** the stack (each interceptor can rewrite it before `proceed()`); the response flows **back up** (each can rewrite it after `proceed()` returns). Because each `proceed()` hands off to a fresh chain at `index + 1`, the structure is a clean one-way descent with a natural unwinding — no shared mutable cursor, which is what makes it safe to short-circuit, retry in a loop, or wrap response bodies.

The key insight: **`proceed()` *is* the chain advancement** — it picks `interceptors[index]`, gives it a chain aimed at `index + 1`, and that interceptor decides whether (and how) to continue.
