`ErrorBoundary.jsx`:

```jsx
import { Component } from 'react';

export class ErrorBoundary extends Component {
  state = { error: null };

  static getDerivedStateFromError(error) {
    return { error };
  }

  componentDidCatch(error, errorInfo) {
    // log error details (swap console for your reporting service)
    console.error('ErrorBoundary caught:', error, errorInfo.componentStack);
    this.props.onError?.(error, errorInfo);
  }

  retry = () => this.setState({ error: null });

  render() {
    if (this.state.error) {
      if (this.props.fallback) {
        return this.props.fallback(this.state.error, this.retry);
      }
      return (
        <div role="alert" style={{ padding: 16 }}>
          <h2>Something went wrong.</h2>
          <pre style={{ whiteSpace: 'pre-wrap' }}>{this.state.error.message}</pre>
          <button onClick={this.retry}>Retry</button>
        </div>
      );
    }
    return this.props.children;
  }
}
```

Usage:

```jsx
<ErrorBoundary onError={(e, info) => sendToSentry(e, info)}>
  <App />
</ErrorBoundary>
```

Notes: must be class (no hook equivalent); `getDerivedStateFromError` sets fallback state, `componentDidCatch` logs. Retry clears error → re-renders children. Optional `fallback(error, retry)` render prop and `onError` callback. To force remount of children on retry, wrap with a `key` bump if children hold bad state.
