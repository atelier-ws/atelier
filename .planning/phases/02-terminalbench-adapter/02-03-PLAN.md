```python
@dataclass
class AdapterResult:
    task_id: str; mode: str; rep: int; model: str

    latency_ms: float; latency_api_ms: float; num_turns: int; cost_usd: float
    grader_verdict: str | None   # "pass" | "fail" | "error" | None
    grader_is_resolved: bool | None
    grader_failure_mode: str

    is_error: bool; stop_reason: str; claude_error: str | None
    stream_log_path: str | None
    atelier_bench_mode: str; atelier_root: str
    dataset_name: str; dataset_version: str

    def to_dict(self) -> dict[str, Any]: ...

def parse_stream_jsonl(log_path: Path) -> dict[str, Any]: ...
# Returns dict with keys: input_tokens, output_tokens, cache_creation_input_tokens,
# cache_read_input_tokens, total_cost_usd, duration_ms, duration_api_ms,
# num_turns, is_error, stop_reason
# Returns {"error": "no_result_line"} when no result line found

class AtelierClaudeAgent(AbstractInstalledAgent):
    def __init__(self, bench_mode: str = "on", model: str | None = None, **kwargs): ...
    @property def _env(self) -> dict[str, str]: ...
    def _run_agent_commands(self, instruction: str) -> list[TerminalCommand]: ...
```
